.. _aws.batch-compute.actions.delete:

delete
++++++
Delete an AWS batch compute environment

:example:

.. code-block:: yaml

    policies:
      - name: delete-environments
        resource: batch-compute
        filters:
          - computeResources.desiredvCpus: 0
        actions:
          - type: delete

    - type


.. _aws.batch-compute.actions.update-environment:

update-environment
++++++++++++++++++
Updates an AWS batch compute environment

:example:

.. code-block:: yaml

    policies:
      - name: update-environments
        resource: batch-compute
        filters:
          - computeResources.desiredvCpus: 0
          - state: ENABLED
        actions:
          - type: update-environment
            state: DISABLED

.. container:: toggle

  

  .. raw:: html
     
    <div class="header docutils container" style=""></div>

  .. code-block:: yaml

    properties:
      computeEnvironment:
        type: string
      computeResources:
        additionalProperties: false
        properties:
          desiredvCpus:
            type: integer
          maxvCpus:
            type: integer
          minvCpus:
            type: integer
        type: object
      serviceRole:
        type: string
      state:
        enum:
        - ENABLED
        - DISABLED
        type: string
      type:
        enum:
        - update-environment
